function login() { let username = document.getElementById("loginUsername").value.trim(); let password = document.getElementById("loginPassword").value.trim(); let error = document.getElementById("loginError"); if (!username || !password) { error.textContent = "Please fill all fields."; return; } let storedPassword = localStorage.getItem(username); if (storedPassword === password) { // Redirect with username parameter window.location.href = `next.html?username=${encodeURIComponent(username)}`; } else { error.textContent = "Invalid username or password."; } }
